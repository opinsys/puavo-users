def acquire(env)
  return if not env["HTTP_AUTHORIZATION"]
  type, data = env["HTTP_AUTHORIZATION"].split(" ")
  if type == "Basic"
    plain = Base64.decode64(data)
    username, password = plain.split(":")
    {
      :username => username,
      :password => password
    }
  end
end
