def all
  a = []
  @store.transaction(true) do
    @store.roots.each do |k|
      a.push(@store[k])
    end
  end
  a
end

def all_without_old
  all.select do |server|
    Time.now - server[:updated] < MAX_AGE
  end
end

def set(domain, attrs)
  attrs[:updated] = Time.now
  attrs[:domain] = domain
  @store.transaction do
    @store[domain] = attrs
  end
end
